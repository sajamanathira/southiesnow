PROCEDURE DIVISION.
    ACCEPT WS-EMP-ID.
    EXEC SQL
    SELECT EMP_ID, EMP_NAME, EMP_LOC, EMP_DOB, EMP_TECH,
       EMP_EARN, EMP_DEDN
    INTO :EMP-ID, :EMP-NAME, :EMP-LOC, :EMP-DOB, :EMP-TECH
       :EMP-EARN, :EMP-DEDN
    FROM T01_EMPLOYEE_TABLE
    WHERE EMP_ID = :WS-EMP-ID
    END-EXEC.
